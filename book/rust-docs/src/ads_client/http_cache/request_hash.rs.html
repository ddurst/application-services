5use std::collections::hash_map::DefaultHasher;
6use std::hash::{Hash, Hasher};
7use viaduct::Request;
8
9#[derive(Clone, Debug, PartialEq)]
10pub struct RequestHash(String);
11
12impl From<&Request> for RequestHash {
13    fn from(request: &Request) -> Self {
14        let mut hasher = DefaultHasher::new();
15        request.method.hash(&mut hasher);
16        request.url.hash(&mut hasher);
17
18        let mut headers: Vec<_> = request.headers.clone().into_iter().collect();
19        headers.sort_by_key(|header| header.name.to_ascii_lowercase());
20        for header in headers {
21            header.name.hash(&mut hasher);
22            header.value.hash(&mut hasher);
23        }
24
25        request.body.hash(&mut hasher);
26        RequestHash(format!("{:x}", hasher.finish()))
27    }
28}
29
30impl From<&str> for RequestHash {
31    fn from(s: &str) -> Self {
32        RequestHash(s.to_string())
33    }
34}
35
36impl From<String> for RequestHash {
37    fn from(s: String) -> Self {
38        RequestHash(s)
39    }
40}
41
42impl std::fmt::Display for RequestHash {
43    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
44        write!(f, "{}", self.0)
45    }
46}

48#[cfg(test)]
49mod tests {
50    use super::*;
51    use viaduct::{Headers, Method, Request};
52
53    fn create_test_request(url: &str, body: &[u8]) -> Request {
54        Request {
55            method: Method::Get,
56            url: url.parse().unwrap(),
57            headers: Headers::new(),
58            body: Some(body.to_vec()),
59        }
60    }
61
62    #[test]
63    fn test_request_hashing() {
64        let request1 = create_test_request("https://example.com/api1", b"body1");
65        let request2 = create_test_request("https://example.com/api2", b"body2");
66        let request3 = create_test_request("https://example.com/api", b"body");
67
68        let hash1 = RequestHash::from(&request1);
69        let hash2 = RequestHash::from(&request2);
70        let hash3 = RequestHash::from(&request3);
71
72        assert_ne!(hash1.to_string(), hash2.to_string());
73        assert_ne!(hash1.to_string(), hash3.to_string());
74        assert_ne!(hash2.to_string(), hash3.to_string());
75
76        let same_request = create_test_request("https://example.com/api", b"body");
77        let hash4 = RequestHash::from(&same_request);
78        assert_eq!(hash3.to_string(), hash4.to_string());
79    }

81    #[test]
82    fn test_request_hashing_header_order_and_case() {
83        let base_url = "https://example.com/api";
84        let body = b"body";
85
86        let req_base = create_test_request(base_url, body);
87        let mut h1 = Headers::new();
88        h1.insert("Accept", "text/plain").unwrap();
89        h1.insert("X-Test", "1").unwrap();
90        let req1 = Request {
91            headers: h1,
92            ..req_base.clone()
93        };
94
95        let req_base = create_test_request(base_url, body);
96        let mut h2 = Headers::new();
97        h2.insert("X-Test", "1").unwrap();
98        h2.insert("Accept", "text/plain").unwrap();
99        let req2 = Request {
100            headers: h2,
101            ..req_base.clone()
102        };
103
104        let req_base = create_test_request(base_url, body);
105        let mut h3 = Headers::new();
106        h3.insert("accept", "text/plain").unwrap();
107        h3.insert("x-test", "1").unwrap();
108        let req3 = Request {
109            headers: h3,
110            ..req_base
111        };
112
113        let h_req1 = RequestHash::from(&req1);
114        let h_req2 = RequestHash::from(&req2);
115        let h_req3 = RequestHash::from(&req3);
116
117        assert_eq!(h_req1.to_string(), h_req2.to_string());
118        assert_eq!(h_req1.to_string(), h_req3.to_string());
119    }

121    #[test]
122    fn test_request_hash_from_string() {
