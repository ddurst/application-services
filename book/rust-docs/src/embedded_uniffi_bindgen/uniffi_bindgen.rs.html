15/// Enumeration of all foreign language targets currently supported by our CLI.
16///
17#[derive(Copy, Clone, Eq, PartialEq, Hash, ValueEnum)]
18enum TargetLanguage {
19    Kotlin,
20    Swift,
21    Python,
22    Ruby,
23}
24
25impl fmt::Display for TargetLanguage {
26    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
27        match self {
28            Self::Kotlin => write!(f, "kotlin"),
29            Self::Swift => write!(f, "swift"),
30            Self::Python => write!(f, "python"),
31            Self::Ruby => write!(f, "ruby"),
32        }
33    }
34}
35
36impl TryFrom<&str> for TargetLanguage {
37    type Error = anyhow::Error;
38    fn try_from(value: &str) -> Result<Self> {
39        Ok(match value.to_ascii_lowercase().as_str() {
40            "kotlin" | "kt" | "kts" => TargetLanguage::Kotlin,
41            "swift" => TargetLanguage::Swift,
42            "python" | "py" => TargetLanguage::Python,
43            "ruby" | "rb" => TargetLanguage::Ruby,
44            _ => bail!("Unknown or unsupported target language: \"{value}\""),
45        })
46    }
47}

49impl TryFrom<&std::ffi::OsStr> for TargetLanguage {
50    type Error = anyhow::Error;
51    fn try_from(value: &std::ffi::OsStr) -> Result<Self> {
52        match value.to_str() {
53            None => bail!("Unreadable target language"),
54            Some(s) => s.try_into(),
55        }
56    }
57}

59impl TryFrom<String> for TargetLanguage {
60    type Error = anyhow::Error;
61    fn try_from(value: String) -> Result<Self> {
62        TryFrom::try_from(value.as_str())
63    }
64}

69/// Scaffolding and bindings generator for Rust
70#[derive(Parser)]
71#[clap(name = "uniffi-bindgen")]
72struct Cli {
73    #[clap(subcommand)]
74    command: Commands,
75}

77#[derive(Subcommand)]
78enum Commands {
79    /// Generate foreign language bindings
80    Generate {
81        /// Foreign language(s) for which to build bindings.
82        #[clap(long, short, value_enum)]
83        language: Vec<TargetLanguage>,
84
85        /// Directory in which to write generated files. Default is same folder as .udl file.
86        #[clap(long, short)]
87        out_dir: Option<Utf8PathBuf>,
88
89        /// Do not try to format the generated bindings.
90        #[clap(long, short)]
91        no_format: bool,
92
93        /// Path to optional uniffi config file. This config is merged with the `uniffi.toml` config present in each crate, with its values taking precedence.
94        #[clap(long, short)]
95        config: Option<Utf8PathBuf>,
96
97        /// Extract proc-macro metadata from a native lib (cdylib or staticlib) for this crate.
98        #[clap(long)]
99        lib_file: Option<Utf8PathBuf>,
100
101        /// Pass in a cdylib path rather than a UDL file
102        #[clap(long = "library")]
103        library_mode: bool,
104
105        /// When `--library` is passed, only generate bindings for one crate.
106        /// When `--library` is not passed, use this as the crate name instead of attempting to
107        /// locate and parse Cargo.toml.
108        #[clap(long = "crate")]
109        crate_name: Option<String>,
110
111        /// Path to the UDL file, or cdylib if `library-mode` is specified
112        source: Utf8PathBuf,
113    },
114
115    /// Generate Rust scaffolding code
116    Scaffolding {
117        /// Directory in which to write generated files. Default is same folder as .udl file.
118        #[clap(long, short)]
119        out_dir: Option<Utf8PathBuf>,
120
121        /// Do not try to format the generated bindings.
122        #[clap(long, short)]
123        no_format: bool,
124
125        /// Path to the UDL file.
126        udl_file: Utf8PathBuf,
127    },
128
129    /// Inspect the bindings render pipeline
130    Pipeline(PipelineArgs),
131}

133#[derive(Args)]
134struct PipelineArgs {
135    /// Pass in a cdylib path rather than a UDL file
136    #[clap(long = "library")]
137    library_mode: bool,
138
139    /// Path to the UDL file, or cdylib if `library-mode` is specified
140    source: Utf8PathBuf,
141
142    /// When `--library` is passed, only generate bindings for one crate.
143    /// When `--library` is not passed, use this as the crate name instead of attempting to
144    /// locate and parse Cargo.toml.
145    #[clap(long = "crate")]
146    crate_name: Option<String>,
147
148    /// Bindings Language
149    language: TargetLanguage,
150
151    /// Only show passes that match <PASS>
152    ///
153    /// Use `last` to only show the last pass, this can be useful when you're writing new pipelines
154    #[clap(short, long)]
155    pass: Option<String>,
156
157    /// Don't show diffs for middle passes
158    #[clap(long)]
159    no_diff: bool,
160
161    /// Only show data for types with name <FILTER_TYPE>
162    #[clap(short = 't', long = "type")]
163    filter_type: Option<String>,
164
165    /// Only show data for items with fields that match <FILTER>
166    #[clap(short = 'n', long = "name")]
167    filter_name: Option<String>,
168}

170fn gen_library_mode(
171    library_path: &camino::Utf8Path,
172    crate_name: Option<String>,
173    languages: Vec<TargetLanguage>,
174    cfo: Option<&camino::Utf8Path>,
175    out_dir: &camino::Utf8Path,
176    fmt: bool,
177) -> anyhow::Result<()> {
178    use uniffi_bindgen::library_mode::generate_bindings;
179
180    let config_supplier = crate::config_supplier::NoCargoConfigSupplier;
181
182    for language in languages {
183        // to help avoid mistakes we check the library is actually a cdylib, except
184        // for swift where static libs are often used to extract the metadata.
185        if !matches!(language, TargetLanguage::Swift) && !uniffi_bindgen::is_cdylib(library_path) {
186            anyhow::bail!(
187                "Generate bindings for {language} requires a cdylib, but {library_path} was given"
188            );
189        }
190
191        // Type-bounds on trait implementations makes selecting between languages a bit tedious.
192        match language {
193            TargetLanguage::Kotlin => generate_bindings(
194                library_path,
195                crate_name.clone(),
196                &KotlinBindingGenerator,
197                &config_supplier,
198                cfo,
199                out_dir,
200                fmt,
201            )?
202            .len(),
203            TargetLanguage::Python => generate_bindings(
204                library_path,
205                crate_name.clone(),
206                &PythonBindingGenerator,
207                &config_supplier,
208                cfo,
209                out_dir,
210                fmt,
211            )?
212            .len(),
213            TargetLanguage::Ruby => generate_bindings(
214                library_path,
215                crate_name.clone(),
216                &RubyBindingGenerator,
217                &config_supplier,
218                cfo,
219                out_dir,
220                fmt,
221            )?
222            .len(),
223            TargetLanguage::Swift => generate_bindings(
224                library_path,
225                crate_name.clone(),
226                &SwiftBindingGenerator,
227                &config_supplier,
228                cfo,
229                out_dir,
230                fmt,
231            )?
232            .len(),
233        };
234    }
235    Ok(())
236}

238fn gen_bindings(
239    udl_file: &camino::Utf8Path,
240    cfo: Option<&camino::Utf8Path>,
241    languages: Vec<TargetLanguage>,
242    odo: Option<&camino::Utf8Path>,
243    library_file: Option<&camino::Utf8Path>,
244    crate_name: Option<&str>,
245    fmt: bool,
246) -> anyhow::Result<()> {
247    use uniffi_bindgen::generate_bindings;
248    for language in languages {
249        match language {
250            TargetLanguage::Kotlin => generate_bindings(
251                udl_file,
252                cfo,
253                KotlinBindingGenerator,
254                odo,
255                library_file,
256                crate_name,
257                fmt,
258            )?,
259            TargetLanguage::Python => generate_bindings(
260                udl_file,
261                cfo,
262                PythonBindingGenerator,
263                odo,
264                library_file,
265                crate_name,
266                fmt,
267            )?,
268            TargetLanguage::Ruby => generate_bindings(
269                udl_file,
270                cfo,
271                RubyBindingGenerator,
272                odo,
273                library_file,
274                crate_name,
275                fmt,
276            )?,
277            TargetLanguage::Swift => generate_bindings(
278                udl_file,
279                cfo,
280                SwiftBindingGenerator,
281                odo,
282                library_file,
283                crate_name,
284                fmt,
285            )?,
286        };
287    }
288    Ok(())
289}

291pub fn run_main() -> anyhow::Result<()> {
292    let cli = Cli::parse();
293    match cli.command {
294        Commands::Generate {
295            language,
296            out_dir,
297            no_format,
298            config,
299            lib_file,
300            source,
301            crate_name,
302            library_mode,
303        } => {
304            if library_mode {
305                if lib_file.is_some() {
306                    panic!("--lib-file is not compatible with --library.")
307                }
308                let out_dir = out_dir.expect("--out-dir is required when using --library");
309                if language.is_empty() {
310                    panic!("please specify at least one language with --language")
311                }
312                gen_library_mode(
313                    &source,
314                    crate_name,
315                    language,
316                    config.as_deref(),
317                    &out_dir,
318                    !no_format,
319                )?;
320            } else {
321                gen_bindings(
322                    &source,
323                    config.as_deref(),
324                    language,
325                    out_dir.as_deref(),
326                    lib_file.as_deref(),
327                    crate_name.as_deref(),
328                    !no_format,
329                )?;
330            }
331        }
332        Commands::Scaffolding {
333            out_dir,
334            no_format,
335            udl_file,
336        } => {
337            uniffi_bindgen::generate_component_scaffolding(
338                &udl_file,
339                out_dir.as_deref(),
340                !no_format,
341            )?;
342        }
343        Commands::Pipeline(args) => {
344            let config_supplier = crate::config_supplier::NoCargoConfigSupplier;
345            let initial_root = if args.library_mode {
346                initial::Root::from_library(config_supplier, &args.source, args.crate_name)?
347            } else {
348                initial::Root::from_udl(config_supplier, &args.source, args.crate_name)?
349            };
350
351            let opts = PrintOptions {
352                pass: args.pass,
353                no_diff: args.no_diff,
354                filter_type: args.filter_type,
355                filter_name: args.filter_name,
356            };
357            match args.language {
358                TargetLanguage::Python => python::pipeline().print_passes(initial_root, opts)?,
359                language => unimplemented!("{language} does not use the bindings IR pipeline yet"),
360            };
361        }
362    };
363    Ok(())
364}
